// ==UserScript== // @name CloudMoon Roblox Customizer (klas) // @namespace http://tampermonkey.net // @version 1.0 // @description Changes title to 'klas', removes ads, and locks Roblox to fullscreen // @author YourName // @match https://*://* // @match https://*://* // @grant none // @run-at document-start // ==/UserScript== (function() { 'use strict'; // 1. Force the title to stay "klas" setInterval(() => { if (document.title !== "klas") { document.title = "klas"; } }, 200); // 2. Clear out ads and promotions const adSelectors = [ '[class*="ad-"]', '[id*="ad-"]', 'iframe[src*="ads"]', '.ad-container', '#ad-banner', '.video-ad', '.google-ads' ]; setInterval(() => { adSelectors.forEach(selector => { document.querySelectorAll(selector).forEach(el => el.remove()); }); }, 500); // 3. Force Fullscreen on first click inside the page window.addEventListener('click', () => { const gameCanvas = document.querySelector('canvas') || document.querySelector('#game-container') || document.documentElement; if (gameCanvas && !document.fullscreenElement) { gameCanvas.requestFullscreen().catch(err => { console.log("Fullscreen blocked: " + err.message); }); } }, { once: true }); // Only runs on the very first click to avoid annoying loops })();
